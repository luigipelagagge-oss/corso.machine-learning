def calcola_batteria(corrente_mA, resistenza_kOhm): """ Input: Corrente in mA, Resistenza in kOhm Output: Tensione in Volt """ # Calcolo la tensione (Nota: i "kilo" e i "milli" si annullano) tensione = corrente_mA * resistenza_kOhm # Restituisco il risultato (Pin di uscita) return tensione
