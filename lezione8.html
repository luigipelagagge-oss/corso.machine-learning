import numpy as np # Parametri del circuito Vin = 5.0 # Tensione di alimentazione R = 1000 # Resistenza in Ohm # Creiamo l'asse delle tensioni per il diodo V_axis = np.linspace(0, 5, 500).reshape(-1, 1) # Retta di Carico: I = (Vin - V) / R I_load = (Vin - V_axis) / R print("Retta di carico calcolata: I = (5 - V) / 1000")

2. Richiamiamo il Modello Ibrido (Lez 7)

# Supponiamo di aver gi√† il modello_diodo dalla Lezione 7 # (Qui semplifichiamo per chiarezza) # Simulazione del modello polinomiale (sostituire con modello reale) def modello_polinomio(V): """Simula il comportamento del modello polinomiale del diodo""" return 1e-12 * (np.exp(V / 0.026) - 1) # Equazione di Shockley # Previsione grezza del modello I_raw = modello_polinomio(V_axis) # APPLICAZIONE DEL FILTRO FISICO (Lezione 7): I_hybrid = np.where(V_axis < 0.6, 0, I_raw) print("Modello ibrido applicato con filtro fisico")

3. Troviamo l'Intersezione (Punto Q)

# Troviamo l'intersezione tra retta di carico e curva del diodo # Cerchiamo il punto dove la differenza √® minima differenza = np.abs(I_hybrid - I_load) idx = np.argmin(differenza) # Indice del punto di minima differenza V_Q = V_axis[idx][0] # Tensione al punto di lavoro I_Q = I_hybrid[idx][0] # Corrente al punto di lavoro print(f"Punto di Lavoro Reale: {V_Q:.2f} V, {I_Q*1000:.2f} mA")
